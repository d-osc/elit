16 // Helper function to render VNode to HTML string (simplified SSR)
17 function renderToString(vNode: VNode | string | number | undefined | null): string {
18 if (vNode == null || vNode === false) return '';
19 if (typeof vNode !== 'object') return String(vNode);
20
21 const { tagName, props, children } = vNode;
22 const attrs = props ? Object.entries(props)
23 .filter(([k, v]) => v != null && v !== false && k !== 'children' && k !== 'ref' && !k.startsWith('on'))
24 .map(([k, v]) => {
25 if (k === 'className' || k === 'class') return `class="${Array.isArray(v) ? v.join(' ') : v}"`;
26 if (k === 'style') return `style="${typeof v === 'string' ? v : Object.entries(v).map(([sk, sv]) => `${sk.replace(/([A-Z])/g, '-$1').toLowerCase()}:${sv}`).join(';')}"`;
27 if (v === true) return k;
28 return `${k}="${v}"`;
29 })
30 .join(' ') : '';
31
32 const childrenStr = children && children.length > 0
33 ? children.map(c => renderToString(c as any)).join('')
34 : '';
35
36 return `<${tagName}${attrs ? ' ' + attrs : ''}>${childrenStr}</${tagName}>`;
37 }
