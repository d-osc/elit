11 // Helper function to render VNode to HTML string (simplified SSR)
12 function renderToString(vNode: VNode | string | number | undefined | null): string {
13 if (vNode == null || vNode === false) return '';
14 if (typeof vNode !== 'object') return String(vNode);
15
16 const { tagName, props, children } = vNode;
17 const attrs = props ? Object.entries(props)
18 .filter(([k, v]) => v != null && v !== false && k !== 'children' && k !== 'ref' && !k.startsWith('on'))
19 .map(([k, v]) => {
20 if (k === 'className' || k === 'class') return `class="${Array.isArray(v) ? v.join(' ') : v}"`;
21 if (k === 'style') return `style="${typeof v === 'string' ? v : Object.entries(v).map(([sk, sv]) => `${sk.replace(/([A-Z])/g, '-$1').toLowerCase()}:${sv}`).join(';')}"`;
22 if (v === true) return k;
23 return `${k}="${v}"`;
24 })
25 .join(' ') : '';
26
27 const childrenStr = children && children.length > 0
28 ? children.map(c => renderToString(c as any)).join('')
29 : '';
30
31 return `<${tagName}${attrs ? ' ' + attrs : ''}>${childrenStr}</${tagName}>`;
32 }
