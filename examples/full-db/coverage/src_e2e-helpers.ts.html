57 /**
58 * Render VNode to HTML string
59 */
60 export function renderToString(vNode: VNode | string | number | undefined | null): string {
61 if (vNode == null || vNode === false) return '';
62 if (typeof vNode !== 'object') return String(vNode);
63
64 const { tagName, props, children } = vNode;
65 const attrs = props ? Object.entries(props)
66 .filter(([k, v]) => v != null && v !== false && k !== 'children' && k !== 'ref' && !k.startsWith('on'))
67 .map(([k, v]) => {
68 if (k === 'className' || k === 'class') return `class="${Array.isArray(v) ? v.join(' ') : v}"`;
69 if (k === 'style') return `style="${typeof v === 'string' ? v : Object.entries(v).map(([sk, sv]) => `${sk.replace(/([A-Z])/g, '-$1').toLowerCase()}:${sv}`).join(';')}"`;
70 if (v === true) return k;
71 return `${k}="${v}"`;
72 })
73 .join(' ') : '';
74
75 const childrenStr = children && children.length > 0
76 ? children.map(c => renderToString(c as any)).join('')
77 : '';
78
79 return `<${tagName}${attrs ? ' ' + attrs : ''}>${childrenStr}</${tagName}>`;
80 }

214 export function parseURL(url: string): { protocol: string; host: string; path: string; query: Record<string, string> } {
215 try {
216 const urlObj = new URL(url);
217 const query: Record<string, string> = {};
218 urlObj.searchParams.forEach((value, key) => {
219 query[key] = value;
220 });
221 return {
222 protocol: urlObj.protocol,
223 host: urlObj.host,
224 path: urlObj.pathname,
225 query
226 };
227 } catch {
228 return { protocol: '', host: '', path: url, query: {} };
229 }

232 /**
233 * Build URL from parts
234 */
235 export function buildURL(parts: {
236 protocol?: string;
237 host?: string;
238 path?: string;
239 query?: Record<string, string>;
240 }): string {
241 const protocol = parts.protocol || 'https:';
242 const host = parts.host || '';
243 const path = parts.path || '/';
244 const query = parts.query || {};
245
246 const queryString = Object.keys(query).length > 0
247 ? '?' + Object.entries(query).map(([k, v]) => `${k}=${encodeURIComponent(v)}`).join('&')
248 : '';
249
250 return `${protocol}//${host}${path}${queryString}`;
251 }

436 /**
437 * Validate HTML structure
438 */
439 export function validateHTML(html: string): {
440 valid: boolean;
441 errors: string[];
442 } {
443 const errors: string[] = [];
444
445 // Check for properly closed tags
446 const voidElements = ['area', 'base', 'br', 'col', 'embed', 'hr', 'img', 'input', 'link', 'meta', 'param', 'source', 'track', 'wbr'];
447 const tagStack: string[] = [];
448 const tagRegex = /<\/?([a-z0-9]+)[^>]*>/gi;
449 let match;
450
451 while ((match = tagRegex.exec(html)) !== null) {
452 const fullTag = match[0];
453 const isClosing = fullTag.startsWith('</');
454 const isSelfClosing = fullTag.endsWith('/>');
455
456 if (isClosing) {
457 const tagName = fullTag.slice(2, -1).toLowerCase();
458 if (tagStack.length === 0 || tagStack[tagStack.length - 1] !== tagName) {
459 // Check if it's a void element (self-closing is allowed)
460 if (!voidElements.includes(tagName)) {
461 errors.push(`Unexpected closing tag: </${tagName}>`);
462 }
463 } else {
464 tagStack.pop();
465 }
466 } else if (isSelfClosing) {
467 const tagName = fullTag.slice(1, -2).toLowerCase();
468 if (!voidElements.includes(tagName)) {
469 errors.push(`Invalid self-closing tag: <${tagName} />`);
470 }
471 } else {
472 const tagName = fullTag.slice(1, fullTag.indexOf(' ') > -1 ? fullTag.indexOf(' ') : fullTag.indexOf('>')).toLowerCase();
473 if (!voidElements.includes(tagName)) {
474 tagStack.push(tagName);
475 }
476 }
477 }
478
479 // Check for unclosed tags
480 tagStack.forEach(tag => {
481 errors.push(`Unclosed tag: <${tag}>`);
482 });
483
484 return {
485 valid: errors.length === 0,
486 errors
487 };
488 }
